CREATE DATABASE fiodouros;
USE fiodouros;

CREATE TABLE estoque (
    id_estoque INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(100) NOT NULL,
    quantidade INT NOT NULL,
    tipo VARCHAR(50)
);


CREATE TABLE movimentacoes (
    id_movimentacao INT PRIMARY KEY AUTO_INCREMENT,
    data_movimentacao DATETIME DEFAULT CURRENT_TIMESTAMP,
    id_produto INT NOT NULL,
    tipo ENUM('entrada', 'saida') NOT NULL,
    quantidade INT NOT NULL,
    CONSTRAINT fk_movimentacoes_estoque
        FOREIGN KEY (id_produto)
        REFERENCES estoque(id_estoque)
);

CREATE TABLE clientes (
    id_cliente INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    telefone VARCHAR(20),
    email VARCHAR(100),
    endereco VARCHAR(150)
);

CREATE TABLE cadastro_servicos (
    id_servico INT AUTO_INCREMENT PRIMARY KEY,
    nome_servico VARCHAR(100) NOT NULL,
    descricao VARCHAR(200),
    valor DECIMAL(10,2) NOT NULL,
    status VARCHAR(20)
);

CREATE TABLE pedidos_servicos (
    id_pedido_servico INT AUTO_INCREMENT PRIMARY KEY,
    id_cliente INT NOT NULL,
    id_servico INT NOT NULL,
    quantidade_pedido INT NOT NULL,
    data_pedido DATETIME DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_pedido_cliente
        FOREIGN KEY (id_cliente)
        REFERENCES clientes(id_cliente),

    CONSTRAINT fk_pedido_servico
        FOREIGN KEY (id_servico)
        REFERENCES cadastro_servicos(id_servico)
);
